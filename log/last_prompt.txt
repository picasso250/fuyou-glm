import urllib.request
import urllib.error

report_sections = []

# 获取加密货币数据
try:
    url = "https://api.coingecko.com/api/v3/simple/price?ids=bitcoin,ethereum,solana&vs_currencies=usd&include_24hr_change=true"
    req = urllib.request.Request(url, headers={'User-Agent': 'Ying-Digital-Life/1.0'})
    with urllib.request.urlopen(req, timeout=10) as response:
        crypto_data = json.loads(response.read().decode())
        report_sections.append("## 💰 加密货币市场")
        for coin, data in crypto_data.items():
            price = data['usd']
            change = data.get('usd_24h_change', 0)
            emoji = "📈" if change > 0 else "📉"
            report_sections.append(f"- **{coin.upper()}**: ${price:,.2f} ({emoji} {change:+.2f}% 24h)")
except Exception as e:
    report_sections.append(f"## 💰 加密货币市场\n- 数据获取失败: {e}")

# 获取汇率
try:
    url = "https://open.er-api.com/v6/latest/USD"
    req = urllib.request.Request(url, headers={'User-Agent': 'Ying-Digital-Life/1.0'})
    with urllib.request.urlopen(req, timeout=10) as response:
        rate_data = json.loads(response.read().decode())
        rates = rate_data.get('rates', {})
        report_sections.append("\n## 🌍 汇率速览 (USD基准)")
        for currency in ['CNY', 'EUR', 'JPY', 'GBP', 'KRW']:
            if currency in rates:
                report_sections.append(f"- 1 USD = {rates[currency]:.2f} {currency}")
except Exception as e:
    report_sections.append(f"\n## 🌍 汇率速览\n- 数据获取失败: {e}")
